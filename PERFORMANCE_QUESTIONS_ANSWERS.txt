PATTERN 1: SIMPLE SINGLETON (RECOMMENDED)
─────────────────────────────────────────────────────────────────────────────────

Code Implementation:
```csharp
// Program.cs - Add at startup
builder.Services.AddSingleton<AIAgent>(sp =>
{
    var client = sp.GetRequiredService<AIProjectClient>();
    return client.GetAIAgent(
        name: "my-agent",
        tools: toolProvider.GetTools(),
        clientFactory: inner => inner.AsBuilder().UseOpenTelemetry().Build()
    );
});

// Controller - Inject directly
public class ChatController : ControllerBase
{
    private readonly AIAgent _agent; // Cached, not resolved per request!
    
    public ChatController(AIAgent agent)
    {
        _agent = agent; // DI provides singleton instance
    }
    
    [HttpPost("chat")]
    public async Task<IActionResult> Chat([FromBody] ChatRequest request)
    {
        // Use _agent directly - no GetAIAgentAsync call!
        var response = await _agent.RunStreamingAsync(request.Message);
        return Ok(response);
    }
}
```

When to use: Agent updates are rare, you can restart app after publishing
Update process: Deploy new agent → Rolling restart of API
Time saved: 2,005ms per request → 0ms per request


PATTERN 2: TIME-BASED REFRESH (AUTO-UPDATE)
─────────────────────────────────────────────────────────────────────────────────

Code Implementation:
```csharp
public class CachedAgentService
{
    private AIAgent? _cachedAgent;
    private DateTime _lastRefresh = DateTime.MinValue;
    private readonly TimeSpan _cacheDuration = TimeSpan.FromMinutes(10);
    private readonly SemaphoreSlim _refreshLock = new(1, 1);
    private readonly AIProjectClient _client;
    private readonly string _agentName;

    public async Task<AIAgent> GetAgentAsync(CancellationToken ct = default)
    {
        // Fast path: return cached if still fresh
        if (_cachedAgent != null && DateTime.UtcNow - _lastRefresh < _cacheDuration)
            return _cachedAgent;

        // Slow path: refresh needed
        await _refreshLock.WaitAsync(ct);
        try
        {
            if (_cachedAgent != null && DateTime.UtcNow - _lastRefresh < _cacheDuration)
                return _cachedAgent;

            _cachedAgent = await _client.GetAIAgentAsync(_agentName, cancellationToken: ct);
            _lastRefresh = DateTime.UtcNow;
            return _cachedAgent;
        }
        finally
        {
            _refreshLock.Release();
        }
    }
}

// Registration
builder.Services.AddSingleton<CachedAgentService>();
```

When to use: Agent updates happen occasionally, need automatic pickup
Refresh frequency: Every 10 minutes (configurable)
Time saved: 2,005ms → 0ms on 99.9% of requests, 310ms on 0.1%

7. ✅ CANCELLATION TOKEN SUPPORT (Save wasted processing)

Implementation:
```csharp
app.MapPost("/chat/stream", async (
    ChatRequest req,
    AIAgent agent,
    HttpContext ctx,
    CancellationToken ct) => // ✅ Add this
{
    ctx.Response.ContentType = "text/event-stream";
    
    try
    {
        await foreach (var update in agent.RunStreamingAsync(req.Message)
            .WithCancellation(ct)) // ✅ Propagate cancellation
        {
            if (ct.IsCancellationRequested) break;
            
            await ctx.Response.WriteAsync($"data: {update.Text}\n\n", ct);
            await ctx.Response.Body.FlushAsync(ct);
        }
    }
    catch (OperationCanceledException)
    {
        // Client disconnected - cleanup
    }
});
```

Impact:
- Stops agent processing immediately when client disconnects
- Saves 5-50 seconds of wasted computation on abandoned requests
- Reduces costs for long-running queries
